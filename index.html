app.get('/api/orders/:id', async (req, res, next) => {
  try {
    const order = await Orders.findById(req.params.id);
    const customer = await cache.getOrSet(`user:${order.userId}`, () => Users.findById(order.userId));
    res.json({ order, customer });
  } catch (err) {
    next(err);
  }
});
